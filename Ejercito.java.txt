import java.util.ArrayList;
import java.util.Random;

public class Ejercito {
    private String nombre;
    private ArrayList<Soldado> soldados;
    private static final Random random = new Random();

    public Ejercito(String nombre) {
        this.nombre = nombre;
        this.soldados = new ArrayList<>();
        generarSoldados();
    }

    private void generarSoldados() {
        int cantidad = random.nextInt(10) + 1; 
        for (int i = 0; i < cantidad; i++) {
            int tipo = random.nextInt(3);
            String nombreSoldado = nombre + "-Soldado" + i;
            int fila = random.nextInt(10);
            int columna = random.nextInt(10);

            switch (tipo) {
                case 0 -> soldados.add(new Espadachin(nombreSoldado, 3 + random.nextInt(2), fila, columna, random.nextDouble() + 1.0));
                case 1 -> soldados.add(new Arquero(nombreSoldado, 1 + random.nextInt(3), fila, columna, random.nextInt(5) + 5));
                case 2 -> soldados.add(new Caballero(nombreSoldado, 3 + random.nextInt(3), fila, columna, random.nextBoolean()));
            }
        }
    }

    public ArrayList<Soldado> getSoldados() {
        return soldados;
    }

    public String getNombre() {
        return nombre;
    }
}
