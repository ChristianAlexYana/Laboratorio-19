public class Espadachin extends Soldado {
    private double longitudEspada;

    public Espadachin(String nombre, int puntosVida, int fila, int columna, double longitudEspada) {
        super(nombre, puntosVida, fila, columna);
        this.longitudEspada = longitudEspada;
    }

    @Override
    public void atacar() {
        System.out.println(nombre + " ataca con su espada de " + longitudEspada + " metros.");
    }

    public void crearMuroEscudos() {
        System.out.println(nombre + " crea un muro de escudos como defensa.");
    }
}
