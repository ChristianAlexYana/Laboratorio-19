public class Caballero extends Soldado {
    private String arma; // Espada o Lanza
    private boolean montado;

    public Caballero(String nombre, int puntosVida, int fila, int columna, boolean montado) {
        super(nombre, puntosVida, fila, columna);
        this.montado = montado;
        this.arma = montado ? "Lanza" : "Espada";
    }

    @Override
    public void atacar() {
        if (montado) {
            System.out.println(nombre + " ataca con su " + arma + " mientras est치 montado.");
        } else {
            System.out.println(nombre + " ataca con su " + arma + " mientras est치 desmontado.");
        }
    }

    public void montar() {
        if (!montado) {
            montado = true;
            arma = "Lanza";
            System.out.println(nombre + " ahora est치 montado y cambia su arma a una lanza.");
        }
    }

    public void desmontar() {
        if (montado) {
            montado = false;
            arma = "Espada";
            System.out.println(nombre + " ahora est치 desmontado y cambia su arma a una espada.");
        }
    }
}
